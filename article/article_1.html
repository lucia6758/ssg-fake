# DB생성 ``` CREATE DATABASE `DB이름`; ``` # table 생성 ``` CREATE TABLE `테이블명` ( `칼럼` 타입 조건, `칼럼2` 타입 조건, ); ``` # row 생성 ``` INSERT INTO `테이블명` SET `칼럼1` = 'data', `칼럼2` = 'data'; ``` ``` INSERT INTO `테이블명` (`칼럼1`,`칼럼2) VALUES ('data','data'); ``` # 예시 ``` CREATE DATABASE textBoard; USE textBoard; CREATE TABLE article ( id INT(10) UNSIGNED NOT NULL PRIMARY KEY AUTO_INCREMENT, regDate DATETIME NOT NULL, title CHAR(200) NOT NULL, `body` TEXT NOT NULL, memberId INT(10) UNSIGNED NOT NULL, ); INSERT INTO article SET regDate = NOW(), title = '제목1', `body` = '내용1', memberId = 1; ```
